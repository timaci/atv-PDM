1)import 'dart:math';

void main() {
  // coordenadas do ponto P1
  double x1 = 3.0;
  double y1 = 4.0;

  // coordenadas do ponto P2
  double x2 = 7.0;
  double y2 = 9.0;

  // cálculo da distância entre os pontos
  double distancia = sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));

  // impressão do resultado
  print('A distância entre os pontos P1($x1,$y1) e P2($x2,$y2) é: $distancia');
}

8)import 'dart:io';


void main() {
  // Leitura dos coeficientes do sistema de equações
  print("Digite os valores dos coeficientes a, b, c, d, e e f: ");
  double a = double.parse(stdin.readLineSync()!);
  double b = double.parse(stdin.readLineSync()!);
  double c = double.parse(stdin.readLineSync()!);
  double d = double.parse(stdin.readLineSync()!);
  double e = double.parse(stdin.readLineSync()!);
  double f = double.parse(stdin.readLineSync()!);

  // Cálculo dos valores de x e y
  double x = (c * e - b * f) / (a * e - b * d);
  double y = (a * f - c * d) / (a * e - b * d);

  // Exibição dos resultados
  print("O valor de x é: $x");
  print("O valor de y é: $y");
}
